module Bug;

def Int fpow(Int b, Int exp) = 
  case exp > 0 {
    True  => b * fpow(b,exp-1);
    False => 1;
  };

class C {
  List<Int> powlist2(List<Int> l, Int b){
    List<Int> result = Nil;
    while (l != Nil) {
      Int h = head(l);
      result = Cons(fpow(b,h),result);
      l = tail(l);
    }
    return result;
  }
}
